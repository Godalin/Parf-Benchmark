This is the command line used to obtain the version of `config.h` versioned
in OSCS:

./configure \
  --disable-readline \
  --disable-sechash \
  --disable-phc \
  --disable-pps \
  --disable-ipv6 \
  --disable-rtc \
  --disable-privdrop \
  --disable-asyncdns \
  --disable-forcednsretry \
  --disable-timestamping
